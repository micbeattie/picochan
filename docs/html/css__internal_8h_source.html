86static inline pch_sid_t get_sid(pch_schib_t *schib) {
87 // if we definitely decide to include intparm in the PMCW then
88 // the schib size is 32 bytes so we could easily check the low
89 // 5 bits are all zero as a valid_params_if check too.
90 valid_params_if(PCH_CSS,
91 schib >= &CSS.schibs[0]
92 && schib < &CSS.schibs[PCH_NUM_SCHIBS]);
93
94 return schib - CSS.schibs;
95}

142static inline pch_intcode_t css_make_intcode(pch_schib_t *schib) {
143 pch_intcode_t ic = { 0 }; // all fields zeroes, including cc
144 if (schib) {
145 pch_sid_t sid = get_sid(schib);
146 ic.intparm = schib->pmcw.intparm;
147 ic.sid = sid;
148 ic.flags = pch_pmcw_isc(&schib->pmcw);
149 ic.cc = 1; // cc=1 means intcode stored [sic]
150 }
151
152 return ic;
153}
