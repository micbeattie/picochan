112static inline uint8_t pch_bsize_encode_raw_inline(uint16_t n) {
113 // XXX TODO See if we can just call pch_bsize_encodex_inline
114 // and return the contained pch_bsize_t and have gcc
115 // reliably optimise it as well as not calculating the
116 // exact flag in the first place. For now we just spell it
117 // all out again.
118
119 // 0b00nnnnnn - 1 x [0,63] -> 0,1,2,...,63
120 if (n <= 63)
121 return (uint8_t)n;
122
123 // 0b01nnnnnn - 2 x [32,95] -> 64,66,68,...,190
124 if (n <= 191)
125 return (uint8_t)(((n>>1) - 32) | 0x40);
126
127 // 0b10nnnnnn - 8 x [24,87] -> 192,200,208,...,696
128 if (n <= 703)
129 return (uint8_t)(((n>>3) - 24) | 0x80);
130
131 // 0b11nnnnnn - 64 x [11,74] -> 704,768,832,...,4736
132 if (n <= 4736)
133 return (uint8_t)(((n>>6) - 11) | 0xc0);
134
135 return 0xff;
136}

149static inline pch_bsizex_t pch_bsize_encodex_inline(uint16_t n) {
150 // 0b00nnnnnn - 1 x [0,63] -> 0,1,2,...,63
151 if (n <= 63)
152 return (pch_bsizex_t){1,pch_bsize_wrap(n)};
153
154 // 0b01nnnnnn - 2 x [32,95] -> 64,66,68,...,190
155 if (n <= 191) {
156 uint8_t exact = (n & 0x1) == 0;
157 pch_bsize_t bsize = pch_bsize_wrap(((n>>1) - 32) | 0x40);
158 return (pch_bsizex_t){exact,bsize};
159 }
160
161 // 0b10nnnnnn - 8 x [24,87] -> 192,200,208,...,696
162 if (n <= 703) {
163 uint8_t exact = (n & 0x7) == 0;
164 pch_bsize_t bsize = pch_bsize_wrap(((n>>3) - 24) | 0x80);
165 return (pch_bsizex_t){exact,bsize};
166 }
167
168 // 0b11nnnnnn - 64 x [11,74] -> 704,768,832,...,4736
169 if (n <= 4736) {
170 uint8_t exact = (n & 0x3f) == 0;
171 pch_bsize_t bsize = pch_bsize_wrap(((n>>6) - 11) | 0xc0);
172 return (pch_bsizex_t){exact,bsize};
173 }
174
175 return (pch_bsizex_t){0, pch_bsize_wrap(0xff)};
176}

195static inline uint16_t pch_bsize_decode_raw_inline(uint8_t esize) {
196 uint8_t flags = esize & 0xc0;
197 uint16_t n = esize & 0x3f;
198
199 switch (flags) {
200 case 0x00:
201 // 0b00nnnnnn - 1 x [0,63] -> 0,1,2,...,63
202 return n;
203
204 case 0x40:
205 // 0b01nnnnnn - 2 x [32,95] -> 64,66,68,...,190
206 return (n+32) << 1;
207
208 case 0x80:
209 // 0b10nnnnnn - 8 x [24,87] -> 192,200,208,...,696
210 return (n+24) << 3;
211 }
212
213 // 0b11nnnnnn - 64 x [11,74] -> 704,768,832,...,4736
214 return (n+11) << 6;
215}
