206/*
207 * \brief Configure an explicit DMA IRQ for use by CUs started from the
208 * calling core and set an exclusive IRQ handler for it.
209 * \ingroup picochan_cu
210 *
211 * If a CSS is to be used on the same Pico, it must be initialised on
212 * a different core, using a different DMA IRQ index. A convenient way
213 * to still allow the CU subsystem to auto-configure its DMA IRQ
214 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
215 * index of the CSS.
216*/
217void pch_cus_configure_dma_irq_index_exclusive(pch_dma_irq_index_t dmairqix);

219/*
220 * \brief Configure an explicit DMA IRQ for use by CUs started from
221 * the calling core and add a shared IRQ handler for it.
222 * \ingroup picochan_cu
223 *
224 * If a CSS is to be used on the same Pico, it must be initialised on
225 * a different core, using a different DMA IRQ index. A convenient way
226 * to still allow the CU subsystem to auto-configure its DMA IRQ
227 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
228 * index of the CSS.

230void pch_cus_configure_dma_irq_index_shared(pch_dma_irq_index_t dmairqix, uint8_t order_priority);

232/*
233 * \brief Configure an explicit DMA IRQ for use by CUs started from
234 * the calling core and add a shared IRQ handler for it using an
235 * order_priority of PICO_SHARED_IRQ_HANDLER_DEFAULT_ORDER_PRIORITY.
236 * \ingroup picochan_cu
237 *
238 * If a CSS is to be used on the same Pico, it must be initialised on
239 * a different core, using a different DMA IRQ index. A convenient way
240 * to still allow the CU subsystem to auto-configure its DMA IRQ
241 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
242 * index of the CSS.
243*/
244void pch_cus_configure_dma_irq_index_shared_default(pch_dma_irq_index_t dmairqix);

246/*
247 * \brief Automatically choose and configure a suitable DMA IRQ for
248 * use by CUs started from the calling core.
249 * \ingroup picochan_cu
250 *
251 * If one of the explicit pch_cus_configure_dma_irq_index_...()
252 * family of functions has already been called from the calling core
253 * then the lowest such DMA IRQ index is returned. Otherwise, the
254 * lowest DMA IRQ index is chosen that has not already been either
255 * configured to any core or explicitly marked as not-to-use by
256 * pch_cus_ignore_dma_irq_index_t(). It is then configured with
257 * pch_cus_configure_dma_irq_index_shared_default() and returned.
258 * In the case that no such unused index is available, the function
259 * panics if required is true, otherwise -1 is returned.
260 *
261 * If a CSS is to be used on the same Pico, it must be initialised on
262 * a different core, using a different DMA IRQ index. A convenient way
263 * to still allow the CU subsystem to auto-configure its DMA IRQ
264 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
265 * index of the CSS.
266*/
267pch_dma_irq_index_t pch_cus_auto_configure_dma_irq_index(bool required);
268
269/* \brief Marks dmairqix such that any call to
270 * pch_cus_auto_configure_dma_irq_index(), whether explicit or
271 * implicitly from pch_cu_start(), will not choose that DMA IRQ index.
272 * \ingroup picochan_cu
273 *
274 * This function is convenient for avoiding the need to configure
275 * explicit DMA IRQ index numbers for the CU subsystem while ensuring
276 * that its auto-configuration of DMA IRQ index numbers does not
277 * conflict with those of a CSS in use on the same Pico or just some
278 * other DMA IRQ index that needs to be reserved for application use.
279 */
280void pch_cus_ignore_dma_irq_index_t(pch_dma_irq_index_t dmairqix);

422static inline pch_unit_addr_t pch_dev_range_get_ua(pch_dev_range_t *dr, uint i) {
423 assert(dr->cu);
424 assert(i < dr->num_devices);
425 assert((uint)dr->first_ua + i < dr->cu->num_devibs);
426
427 return dr->first_ua + i;
428}

430static inline pch_unit_addr_t pch_dev_range_get_ua_required(pch_dev_range_t *dr, uint i) {
431 if (!dr->cu)
432 panic("missing cu in dev_range");
433
434 if (i >= dr->num_devices)
435 panic("index not in dev_range");
436
437 return dr->first_ua + i;
438}

440static inline int pch_dev_range_get_index_nocheck(pch_dev_range_t *dr, pch_devib_t *devib) {
441 return (int)pch_dev_get_ua(devib) - dr->first_ua;
442}

444static inline int pch_dev_range_get_index(pch_dev_range_t *dr, pch_devib_t *devib) {
445 assert(dr->cu == pch_dev_get_cu(devib));
446
447 int i = pch_dev_range_get_index_nocheck(dr, devib);
448 if (i < 0 || i >= dr->num_devices)
449 return -1;
450
451 return i;
452}

454static inline int pch_dev_range_get_index_required(pch_dev_range_t *dr, pch_devib_t *devib) {
455 int i = pch_dev_range_get_index(dr, devib);
456 if (i < 0)
457 panic("devib not found in dev_range");
