256/*
257 * \brief Configure an explicit DMA IRQ for use by CUs started from the
258 * calling core and set an exclusive IRQ handler for it.
259 * \ingroup picochan_cu
260 *
261 * If a CSS is to be used on the same Pico, it must be initialised on
262 * a different core, using a different DMA IRQ index. A convenient way
263 * to still allow the CU subsystem to auto-configure its DMA IRQ
264 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
265 * index of the CSS.
266*/

272 * \ingroup picochan_cu

296/*
297 * \brief Automatically choose and configure a suitable DMA IRQ for
298 * use by CUs started from the calling core.
299 * \ingroup picochan_cu
300 *
301 * If one of the explicit pch_cus_configure_dma_irq_index_...()
302 * family of functions has already been called from the calling core
303 * then the lowest such DMA IRQ index is returned. Otherwise, the
304 * lowest DMA IRQ index is chosen that has not already been either
305 * configured to any core or explicitly marked as not-to-use by
306 * pch_cus_ignore_dma_irq_index_t(). It is then configured with
307 * pch_cus_configure_dma_irq_index_shared_default() and returned.
308 * In the case that no such unused index is available, the function
309 * panics if required is true, otherwise -1 is returned.
310 *
311 * If a CSS is to be used on the same Pico, it must be initialised on
312 * a different core, using a different DMA IRQ index. A convenient way
313 * to still allow the CU subsystem to auto-configure its DMA IRQ
314 * choice is to call pch_cus_ignore_dma_irq_index_t() on the DMA IRQ
315 * index of the CSS.
316*/
317pch_dma_irq_index_t pch_cus_auto_configure_dma_irq_index(bool required);
318
319/* \brief Marks dmairqix such that any call to
320 * pch_cus_auto_configure_dma_irq_index(), whether explicit or
321 * implicitly from pch_cu_start(), will not choose that DMA IRQ index.
322 * \ingroup picochan_cu
323 *
324 * This function is convenient for avoiding the need to configure
325 * explicit DMA IRQ index numbers for the CU subsystem while ensuring
326 * that its auto-configuration of DMA IRQ index numbers does not
327 * conflict with those of a CSS in use on the same Pico or just some
328 * other DMA IRQ index that needs to be reserved for application use.
329 */
330void pch_cus_ignore_dma_irq_index_t(pch_dma_irq_index_t dmairqix);

505static inline pch_unit_addr_t pch_dev_range_get_ua_required(pch_dev_range_t *dr, uint i) {
506 if (!dr->cu)
507 panic("missing cu in dev_range");
508
509 if (i >= dr->num_devices)
510 panic("index %lu not in dev_range", (unsigned long)i);
511
512 return dr->first_ua + i;
513}
