195static inline bool pch_cbindex_is_registered(uint cbindex) {
196 if (cbindex >= NUM_DEVIB_CALLBACKS)
197 return false;

202static inline bool pch_cbindex_is_callable(uint cbindex) {
203 if (cbindex == PCH_DEVIB_CALLBACK_NOOP)
204 return true;
205
206 if (cbindex >= NUM_DEVIB_CALLBACKS)
207 return false;
208
209 return pch_devib_callbacks[cbindex].func != NULL;
210}
