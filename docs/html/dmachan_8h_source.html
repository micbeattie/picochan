45static inline bool dma_irqn_get_channel_forced(uint irq_index, uint channel) {
46 invalid_params_if(HARDWARE_DMA, irq_index >= NUM_DMA_IRQS);
47 check_dma_channel_param(channel);
48
49 return dma_hw->irq_ctrl[irq_index].intf & (1u << channel);
50}
51
52static inline void dma_irqn_set_channel_forced(uint irq_index, uint channel, bool forced) {
53 invalid_params_if(HARDWARE_DMA, irq_index >= NUM_DMA_IRQS);
54
55 if (forced)
56 hw_set_bits(&dma_hw->irq_ctrl[irq_index].intf, 1u << channel);
57 else
58 hw_clear_bits(&dma_hw->irq_ctrl[irq_index].intf, 1u << channel);
59}
