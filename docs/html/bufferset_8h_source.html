32// pch_trc_switch_to_next_buffer_unsafe is for internal use.
33// The external API is pch_trc_switch_to_next_buffer which takes
34// the trace_lock and then calls this with a 0 for position.
35// Internally, this is used when allocating a slot for a new trace
36// record (which has already taken trace_lock) and in that situation
37// it is often called with a non-zero pos following the
38// newly-allocated trace record.
39static inline unsigned char *pch_trc_switch_to_next_buffer_unsafe(pch_trc_bufferset_t *bs, uint32_t pos) {
40 bs->current_buffer_num = (bs->current_buffer_num + 1) % PCH_TRC_NUM_BUFFERS;
41 bs->current_buffer_pos = pos;
42 if (bs->irqnum > -1)
43 irq_set_pending((irq_num_t)(bs->irqnum));
44
45 return bs->buffers[bs->current_buffer_num];
46}
47
48// pch_trc_switch_to_next_buffer switches to the next trace buffer in
49// the bufferset. If bs->irqnum is non-negative, that IRQ is raised.
50// When the IRQ is raised, current_buffer_num has already been
51// incremented (modulo PCH_TRC_NUM_BUFFERS) and a trace record may be
52// in the process of writing to the new buffer. The IRQ handler will
53// typically want to start copying or sending the contents of
54// bs->buffers[bs->current_buffer_num-1] elsewhere and aim for
55// completion before the trace records fill remaining buffers and
56// wrap back around to overwrite that buffer.
57static inline unsigned char *pch_trc_switch_to_next_buffer(pch_trc_bufferset_t *bs) {
58 uint32_t status = trace_lock();
59 unsigned char *rec = pch_trc_switch_to_next_buffer_unsafe(bs, 0);
60 trace_unlock(status);
61 return rec;
62}
