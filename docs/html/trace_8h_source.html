28// pch_trc_write allocates and writes the header of a trace record
29// with the current timestamp and record type rt and returns a pointer
30// to the location where data_size bytes of associated trace should be
31// written. It returns NULL (without writing any header or taking any
32// other action) if no trace record should be written. This will be
33// the case if tracing was disabled globally at compile time
34// (PCH_CONFIG_ENABLE_TRACE was not defined or defined as 0) or if
35// tracing has been disabled (perhaps temporarily) at runtime by
36// setting pch_trc_enable to false or if the cond function argument is
37// false.
38static inline void *pch_trc_write(pch_trc_bufferset_t *bs, bool cond, pch_trc_record_type_t rt, uint8_t data_size) {
39#if PCH_CONFIG_ENABLE_TRACE
40 if (!bs->enable // per-bufferset runtime tracing flag not enabled
41 || !cond) // per-function-call condition flag not enabled
42 return NULL;
43
44 return pch_trc_write_uncond(bs, rt, data_size);
45#else
46 (void)bs;
47 (void)cond;
48 (void)rt;
49 (void)data_size;
50#endif
51 return NULL;
52}

54#define PCH_TRC_WRITE(bs, cond, rt, data) do { \
55 size_t __data_size = sizeof (data); \
56 void *__rec = pch_trc_write((bs), (cond), (rt), __data_size); \
57 if (__rec) \
58 memcpy(__rec, &(data), __data_size); \
59 } while (0)
