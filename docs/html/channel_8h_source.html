21// ua_dlist_t is the head of a circular double-linked list of schibs
22// which all belong to the same channel, linked by the prevua/nextua
23// fields of schib.mda. It is the unit_addr_t of the head of the list
24// or else -1 if the list is empty.

27// ua_slist_t is the head and tail of a single-linked list of schibs
28// which all belong to the same channel, linked by the nextua field of
29// schib.mda. It contains the unit_addr_t of the head and tail of
30// the list or else both fields are -1 if the list is empty.

32 int16_t head;
33 int16_t tail;

243static inline proto_packet_t get_tx_packet(pch_chp_t *chp) {
244 // chp.tx_channel is a dmachan_tx_channel_t which is the
245 // first member of chp which is a pch_chp_t which is
246 // __aligned(4) and cmd is the first member of tx_channel
247 // so is 4-byte aligned. proto_packet_t is 4-bytes and also
248 // __aligned(4) (and needing no more than 4-byte alignment)
249 // but omitting the __builtin_assume_aligned below causes
250 // gcc 14.1.0 to produce error
251 // error: cast increases required alignment of target type
252 // [-Werror=cast-align]
253 proto_packet_t *pp = (proto_packet_t *)
254 __builtin_assume_aligned(&chp->tx_channel.link.cmd, 4);
255 return *pp;
256}
