19static inline bool dma_irqn_get_channel_forced(uint irq_index, uint channel) {
20 invalid_params_if(HARDWARE_DMA, irq_index >= NUM_DMA_IRQS);
21 check_dma_channel_param(channel);
22
23 return dma_hw->irq_ctrl[irq_index].intf & (1u << channel);
24}

26static inline void dma_irqn_set_channel_forced(uint irq_index, uint channel, bool forced) {
27 invalid_params_if(HARDWARE_DMA, irq_index >= NUM_DMA_IRQS);
28
29 if (forced)
30 hw_set_bits(&dma_hw->irq_ctrl[irq_index].intf, 1u << channel);
31 else
32 hw_clear_bits(&dma_hw->irq_ctrl[irq_index].intf, 1u << channel);
33}
