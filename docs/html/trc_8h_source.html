37static inline void pch_trc_write_timestamp(pch_trc_timestamp_t *tp, uint64_t us) {
38 uint16_t *hp = (uint16_t*)tp;
39 hp[0] = (uint16_t)us; // low 16 bits: t.low
40 hp[1] = (uint16_t)(us >> 16); // middle 16 bits: t.mid
41 hp[2] = (uint16_t)(us >> 32); // low 16 bits of top 32: t.high
42}
