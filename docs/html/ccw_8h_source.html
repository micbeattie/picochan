68// Architected values of CCW commands.
69// These do not match those for traditional CSS and we only divide
70// into "Read/Write" via the low bit instead of into Control/Read/
71// ReadBackward/Sense/Test/Write via various low-bit groups.
72
73#define PCH_CCW_CMD_FIRST_RESERVED 0xf0
74// WRITE
75#define PCH_CCW_CMD_WRITE 0x01
76// READ
77#define PCH_CCW_CMD_READ 0x02
78// TIC: Transfer In Channel
79#define PCH_CCW_CMD_TIC 0xf0
80// SENSE: Read Sense data from devib
81#define PCH_CCW_CMD_SENSE 0xf2

83// Architected bit tests of CCW commands
84static inline bool pch_is_ccw_cmd_write(uint8_t cmd) {
85 return (cmd & 0x01) == 1;

88static inline bool pch_is_ccw_cmd_read(uint8_t cmd) {
